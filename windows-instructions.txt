1. Install GTK (https://www.gtk.org/docs/installations/windows/)
 - Especially run:
   pacman -S mingw-w64-x86_64-gtk3
   pacman -S base-devel mingw-w64-x86_64-toolchain
2. Add C:\msys64\mingw64\bin to PATH
3. In CMakeLists.txt change ADD_EXECUTABLE(... src/connection/win_fifo.c ...) to ADD_EXECUTABLE(... src/connection/lin_fifo.c ...)
4. cd resources
   "C:\msys64\mingw64\bin\glib-compile-resources.exe" gresource.xml --target=resources.c --generate-source
5. C:\msys64\mingw64\bin\cmake.exe -DCMAKE_BUILD_TYPE=Debug -DCMAKE_MAKE_PROGRAM=C:/msys64/mingw64/bin/make.exe -DCMAKE_C_COMPILER=C:/msys64/mingw64/bin/gcc.exe -G "CodeBlocks - MinGW Makefiles" C:\Users\Bartek\CLionProjects\Tic-Tac-Toe-C
   C:\msys64\mingw64\bin\cmake.exe --build C:\Users\Bartek\CLionProjects\Tic-Tac-Toe-C --target Tic_Tac_Toe_C -- -j 4
